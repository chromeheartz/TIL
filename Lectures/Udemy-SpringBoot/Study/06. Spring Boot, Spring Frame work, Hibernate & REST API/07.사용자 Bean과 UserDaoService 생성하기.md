	private String name;

	super();
	this.id = id;
	this.name = name;
	this.birthDate = birthDate;
}

	return id;
}

	this.id = id;
}

public String getName() {
	return name;
}

	this.name = name;
}

public LocalDate getBirthDate() {
	return birthDate;
}

public void setBirthDate(LocalDate birthDate) {
	this.birthDate = birthDate;
}

@Override
public String toString() {
	return "User [id=" + id + ", name=" + name + ", birthDate=" + birthDate + "]";
}
