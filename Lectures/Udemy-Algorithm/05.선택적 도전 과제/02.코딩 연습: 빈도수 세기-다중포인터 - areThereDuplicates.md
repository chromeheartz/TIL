function areThereDuplicates(...args) {
  let argsCounter = {};

  for (let val of args) {
    argsCounter[val] = (argsCounter[val] || 0) + 1;
  }
  for (let key in argsCounter) {
    if (argsCounter[key] > 1) {
      return true;
    }
  }
  return false;
}
