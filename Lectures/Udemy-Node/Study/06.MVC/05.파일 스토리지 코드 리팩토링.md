### 📌 파일 스토리지 코드 리팩토링

모델에 파일 저장 논리를 추가해보자.

일단 헬퍼 함수를 생성하고 상수에 저장한다. 이 헬퍼 함수는 경로 구축하고 파일을 읽어온다.
기본적으로 `fetchAll`에서 하는 일을 한다.
`const getProductsFormFile = () => {}`
그 뿐만 아니라 콜백도 받고, 왜냐면 여기서 실행하고 최종적으로 이를 회신하기 때문이다.
이 과정에는 다소 시간이 소요되고 여전히 함수가 완료되었을때 `호출자` 에게 알려야 하기 때문이다.

따라서 같은 방식으로 파일 읽기를 완료하면 콜백을 받아서 실행하는 `헬퍼 함수` 를 사용한다.
이렇게 하면 여기 `fetachAll` 에서 실행하는 것은 사실상 `getProductsFromFile` 이 전부이다.
그러므로 이를 호출해서 콜백을 `포워딩` 한다.

이렇게 되면 `save` 에서도 똑같이 헬퍼 함수를 호출할 수 있음을 뜻한다.
그러나 여기는 자체 논리가 있으므로 콜백을 포워딩하지는 않는다.

```js
// 리팩토링 전
const fs = require("fs");
const path = require("path");

module.exports = class Product {
  constructor(t) {
    this.title = t;
  }

  save() {
    const p = path.join(
      path.dirname(process.mainModule.filename),
      "data",
      "products.json",
    );
    fs.readFile(p, (err, fileContent) => {
      let products = [];
      if (!err) {
        products = JSON.parse(fileContent);
      }
      products.push(this);
      fs.writeFile(p, JSON.stringify(products), (err) => {
        console.log(err);
      });
    });
  }

  static fetchAll(cb) {
    const p = path.join(
      path.dirname(process.mainModule.filename),
      "data",
      "products.json",
    );
    fs.readFile(p, (err, fileContent) => {
      if (err) {
        cb([]);
      }
      cb(JSON.parse(fileContent));
    });
  }
};

// 리팩토링 후
const fs = require("fs");
const path = require("path");

const p = path.join(
  path.dirname(process.mainModule.filename),
  "data",
  "products.json",
);

const getProductsFromFile = (cb) => {
  fs.readFile(p, (err, fileContent) => {
    if (err) {
      cb([]);
    } else {
      cb(JSON.parse(fileContent));
    }
  });
};

module.exports = class Product {
  constructor(t) {
    this.title = t;
  }

  save() {
    getProductsFromFile((products) => {
      products.push(this);
      fs.writeFile(p, JSON.stringify(products), (err) => {
        console.log(err);
      });
    });
  }

  static fetchAll(cb) {
    getProductsFromFile(cb);
  }
};
```
