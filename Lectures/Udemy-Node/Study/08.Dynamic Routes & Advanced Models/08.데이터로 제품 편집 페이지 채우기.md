### 📌 데이터로 제품 편집 페이지 채우기

```js
exports.getEditProduct = (req, res, next) => {
  const editMode = req.query.edit;
  if (!editMode) {
    return res.redirect("/");
  }
  const prodId = req.params.productId;
  Product.findById(prodId, (product) => {
    if (!product) {
      return res.redirect("/");
    }
    res.render("admin/edit-product", {
      pageTitle: "Edit Product",
      path: "/admin/edit-product",
      editing: editMode,
      product: product,
    });
  });
};
```

`edit-product` 의 `form` 양식을 제품 데이터로 채우기 위해 먼저 제품을 불러와야 한다.
`getEditProduct` 에서 동적세그먼트를 총해 제품 ID를 추출해내고, 이를 사용해서 제품 모델을 찾을 수 있다.
그리고 제품이 있는지 확인하는 검사를 추가해서 없다면 사용자를 다른 페이지로 재이동시킨다.
이런 경우에 `오류` 가 나게 하는게 맞지만 일단은 현재로 둔다.

```html
<%- include('../includes/head.ejs') %>
    <link rel="stylesheet" href="/css/forms.css">
    <link rel="stylesheet" href="/css/product.css">
</head>

<body>
   <%- include('../includes/navigation.ejs') %>

    <main>
        <form class="product-form" action="/admin/<% if (editing) { %>edit-product<% } else { %>add-product<% } %>" method="POST">
            <div class="form-control">
                <label for="title">Title</label>
                <input type="text" name="title" id="title" value="<% if (editing) { %><%= product.title %><% } %>">
            </div>
            <div class="form-control">
                <label for="imageUrl">Image URL</label>
                <input type="text" name="imageUrl" id="imageUrl" value="<% if (editing) { %><%= product.imageUrl %><% } %>">
            </div>
            <div class="form-control">
                <label for="price">Price</label>
                <input type="number" name="price" id="price" step="0.01" value="<% if (editing) { %><%= product.price %><% } %>">
            </div>
            <div class="form-control">
                <label for="description">Description</label>
                <textarea name="description" id="description" rows="5"><% if (editing) { %><%= product.description %><% } %></textarea>
            </div>

            <button class="btn" type="submit"><% if (editing) { %>Update Product<% } else { %>Add Product<% } %></button>
        </form>
    </main>
<%- include('../includes/end.ejs') %>
```

`edit-product.ejs` 에서도 조건을 걸어주어서 버튼에 맞게 텍스트가 보이도록 수정해주었다.

이런식으로 사용하게 되면 하나의 ejs를 가지고 `add / edit` 두개의 페이지를 템플릿으로 사용할 수 있게 된다.
`add / edit` 에 각각 바뀌어야 할 부분이나 보여줘야 할 정보를 조건으로 맞춰서 `편집 모드` 인지 확인을 하고 작성해주면 된다.``
